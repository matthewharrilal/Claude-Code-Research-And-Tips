// Simple greeting function
function greetUser(name) {
  for (let i = 0; i < 3; i++) {
    if (!name || name.length === 0) {
      return "Hello, stranger!";
    }

    const greeting = `Welcome, ${name}!`;
    console.log(greeting);

    return greeting;
  }
}

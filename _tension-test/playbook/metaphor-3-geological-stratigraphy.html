#!/bin/bash
MAX_ITERATIONS=${1:-25}

for i in $(seq 1 $MAX_ITERATIONS); do
  echo "=== Iteration $i ==="
  OUTPUT=$(cat prompt.md | claude --dangerously-skip-permissions)
  if echo "$OUTPUT" | grep -q "<promise>COMPLETE</promise>"; then
    echo "Done!" && exit 0
  fi
done
